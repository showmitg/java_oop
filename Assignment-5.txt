public class Employee{
    public String name;
    public int id;
    public String status;
    public Employee(){
        System.out.println("A default employee has been created");
    }
    public Employee(String name, int id, String status){
        this.name = name;
        this.id = id;
        this.status = status;
    }
}

public class Company{
    public String name;
    public Employee[] emp; 
    public int count;
    public int max = 3; 

    public Company(){
        name = "ABC Company";
        emp = new Employee[max];
        count = 0;
    }

    public void addEmployee(Employee e){
        if(count < max){
            emp[count] = e;
            count++;
            if(!e.name.equals("")){
                 System.out.println(e.name + " has joined the company");
            }
        } else {
            System.out.println("No more vacancy");
        }
    }

    public void removeEmployee(Employee e){
        for(int i = 0; i < count; i++){
            if(emp[i].id == e.id){
                emp[i] = null;
                count--;
                break;
            }
        }
    }

    public void details(){
        System.out.println("Company Name: " + name);
        System.out.println("Total Employee: " + count);
        
        System.out.println("Fulltime Employee: ");
        for(int i = 0; i < count; i++){
            if(emp[i].status.equals("Fulltime")){
                System.out.println("Name: " + emp[i].name + ", ID: " + emp[i].id);
            }
        }

        System.out.println("Part-Time Employee: ");
        for(int i = 0; i < count; i++){
            if(emp[i].status.equals("Part-time")){
                System.out.println("Name: " + emp[i].name + ", ID: " + emp[i].id);
            }
        }
    }
}

public class Student {
    public String name;
    public int id;
    public double cgpa;

    public Student(String name, int id, double cgpa) {
        this.name = name;
        this.id = id;
        this.cgpa = cgpa;
    }

    public void updateId(int newId) {
        this.id = newId;
    }
}



public class Department {
    public String name;
    public Student[] list = new Student[5];
    public int count = 0;

    public Department(String name) {
        this.name = name;
    }

    public void addStudent(Student s1, Student s2, Student s3) {
        addSingle(s1);
        addSingle(s2);
        addSingle(s3);
    }

    public void addStudent(Student s) {
        addSingle(s);
    }

    private void addSingle(Student s) {
        if (count == 5) return;
        for (int i = 0; i < count; i++) {
            if (list[i].id == s.id) {
                System.out.println("Student with the same ID already exists, Please try with another ID");
                return;
            }
        }

        list[count] = s;
        count++;

        System.out.println("Welcome to " + name + " department, " + s.name);
    }

    public void findStudent(int id) {

        if (id <= 0) {
            System.out.println("Student with this ID doesn't exist, Please give a valid ID");
            return;
        }

        for (int i = 0; i < count; i++) {
            if (list[i].id == id) {
                System.out.println("Student info:");
                System.out.println("Student Name: " + list[i].name);
                System.out.println("ID: " + list[i].id);
                System.out.println("CGPA:  " + list[i].cgpa);
                return;
            }
        }

        System.out.println("Student with this ID doesn't exist, Please give a valid ID");
    }

    public void details() {
        System.out.println("Department Name: " + name);
        System.out.println("Number of student:" + count);
        System.out.println("Details of the students: ");

        for (int i = 0; i < count; i++) {
            System.out.println("Student name: " + list[i].name + ", ID: " + list[i].id + ", cgpa: " + list[i].cgpa);
        }
    }
}

public class UberEatsUser{
    public String name;
    public String location;
    public boolean pen;

    public UberEatsUser(String n, String l){
        name = n;
        location = l;
        pen = false;
    }

    public UberEatsUser(String n){
        name = n;
        location = "Unknown";
        pen = false;
    }

    public void updateLocation(String newLocation){
        if(location.equals("Unknown")){
            location = newLocation;
            System.out.println("Update Successful!");
        } else if (location.equals(newLocation)){
             System.out.println("We already have your location. Please place an order!");
        } else {
            location = newLocation;
        }
    }
}


public class Restaurant{
    public String name;
    public String loc;
    public UberEatsUser[] orders;
    public int count;
    public int max = 3;

    public Restaurant(String n, String l){
        name = n;
        loc = l;
        orders = new UberEatsUser[max];
        count = 0;
    }

    public void takeOrder(UberEatsUser user){
        if(user.location.equals("Unknown")){
            System.out.println("Location : Unknown.");
            System.out.println("Please update your location information!");
            return;
        }

        if(user.pen){
            System.out.println("You already have a pending order!");
            return;
        }

        if(count < max){
            orders[count] = user;
            user.pen = true;
            count++;
            System.out.println("Your order has been added!");
        } else {
            System.out.println("We are really busy right now.");
            System.out.println("Please order after some time. Thank you!");
        }
    }

    public void completeOrders(){
        if(count == 0){
            System.out.println("No pending orders at the moment.");
            return;
        }

        System.out.println("Showing " + name + "'s orders:");
        for(int i = 0; i < count; i++){
            UberEatsUser user = orders[i];
            System.out.println("Order by " + user.name + " at " + user.location + " completed");
            user.pen = false;
        }
        orders = new UberEatsUser[max];
        count = 0;
    }
}

public class ConnectFriends {
    public String name;
    public int max;
    public String[] sent;
    public int total;

    public ConnectFriends(String name) {
        this.name = name;
        this.max = 2;
        this.sent = new String[max];
        System.out.println("Welcome to ConnectFriends, " + name);
    }

    public ConnectFriends(String name, int max) {
        this.name = name;
        this.max = max;
        this.sent = new String[max];
        System.out.println("Welcome to ConnectFriends, " + name);
    }

    public void sendFriendRequest(ConnectFriends a) {
        if (total < max) {
            sent[total] = a.name;
            total++;
            System.out.println(name + " sent a friend request to " + a.name + ".");
        } else {
            System.out.println(name + " has reached the friend request limit!");
        }
    }

    public void sendFriendRequest(ConnectFriends a, ConnectFriends b) {
        this.sendFriendRequest(a);
        this.sendFriendRequest(b);
    }

    public void removeRequest(String user) {
        int index = -1;
        for (int i = 0; i < total; i++) {
            if (sent[i].equals(user)) {
                index = i;
                break;
            }
        }

        if (index == -1) {
            System.out.println(user + " is not in " + name + "'s sent request list.");
        } else {
            for (int i = index; i < total - 1; i++) {
                sent[i] = sent[i + 1];
            }
            total--;
            System.out.println("Reuqest to add " + user + " is removed for " + name + ".");
        }
    }
    public void showDetails() {
        System.out.println("User Name: " + name);
        System.out.println("Maximum number of Sent Friend Request: " + max);
        System.out.println("Total Friends Request: " + total);
        System.out.print("Sent Friends Request: ");
        for (int i = 0; i < total; i++) {
            System.out.print(sent[i] + " ");
        }
        System.out.println();
    }
}
