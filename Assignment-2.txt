public class Assignment {
    int tasks;
    String difficulty;
    boolean submission;

    public Assignment() {
        tasks = 0;
        difficulty = null;
        submission = false;
    }

    public void printDetails() {
        System.out.println("Number of tasks: " + tasks);
        System.out.println("Difficulty level: " + difficulty);
        System.out.println("Submission required: " + submission);
    }

    public String makeOptional() {
        if (submission) {
            submission = false;
            return "Assignment will not require submission";
        } else {
            return "Submission is already not required";
        }
    }
}

public class Employee {
    public String name;
    public double salary = 30000;
    public String designation = "junior";

    public void newEmployee(String na) {
        name = na;
    }

    public void calculateTax() {
        if (salary > 50000) {
            System.out.println(name + " Tax Amount: " + salary * 0.3);
        } else if (salary > 30000 && salary <= 50000) {
            System.out.println(name + " Tax Amount: " + salary * 0.1);
        } else {
            System.out.println("No need to pay tax");
        }
    }

    public void promoteEmployee(String des) {
        designation = des;
        if (des.equals("senior")) {
            salary += 25000;
        } else if (des.equals("lead")) {
            salary += 50000;
        } else if (des.equals("manager")) {
            salary += 75000;
        }

        System.out.println(name + " has been promoted to " + designation);
        System.out.println("New Salary: " + salary);
    }

    public void displayInfo() {
        System.out.println("Employee Name: " + name);
        System.out.println("Employee Salary: " + salary + " Tk");
        System.out.println("Employee Designation: " + designation);
    }
}

public class MobilePhone {
    public int con;
    public String[] names;
    public int[] numbers;
    public int tcount = 0;

    public void setContactCapacity(int n) {
        con = n;
        names = new String[con];
        numbers = new int[con];
    }

    public void addContact(String name, int number) {
        if (tcount < con) {
            names[tcount] = name;
            numbers[tcount] = number;
            tcount++;
            System.out.println("The contact of " + name + " is added.");
        } else {
            System.out.println("Storage Full!!");
        }
    }

    public void makeCall(int n) {
        boolean found = false;
        for (int i = 0; i < tcount; i++) {
            if (numbers[i] == n) {
                System.out.println("Calling " + names[i] + " . . .");
                found = true;
                break;
            }
        }
        if (!found) {
            System.out.println("Calling " + n + " . . .");
        }
    }

    public void details() {
        System.out.println("Total Contacts: " + tcount);
        System.out.println("Contact List:");
        for (int i = 0; i < tcount; i++) {
            System.out.println(names[i] + ":" + numbers[i]);
        }
    }
}
